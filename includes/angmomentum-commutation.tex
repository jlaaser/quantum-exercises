			\begin{align*}
				\hat L_x = \hat y \hat p_z - \hat z \hat p_y & &
				\hat L_y = \hat z \hat p_x - \hat x \hat p_z & &
				\hat L_z = \hat x \hat p_y - \hat y \hat p_x
